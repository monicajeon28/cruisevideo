##   탐지 (+ 패턴)



  패턴 "() 없는 테스트"
  예시 
    (' ', ()  
         ()
      //  없음!
    )


  패턴 "/ 단독"
  예시 
    //  
    ().()
    ().()

    //  
    ().(  ,  '' )


  패턴 "항상 통과하는 테스트"
  예시 
    //  
    ().()
    ( + ).()


  패턴 "주석 처리된 "
  예시 
    (' ', ()  
         ()
      // (.).()
    )


  패턴 "모든 것을 모킹"
  예시 
    //  의미 없는 테스트
    .('./')
    .('./')
    .('./')
    // 실제 로직 테스트 안 함
