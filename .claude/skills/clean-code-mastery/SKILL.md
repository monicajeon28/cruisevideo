│   ├── .                 #  패턴 + 보안 (줄)
│   ├── .                  # ++ 패턴 + 보안 (줄)
│   ├── .               # # 패턴 + 보안 (줄)
│   └── .               #  패턴 + 보안 (줄)
├── /
│   ├── -.     # 코드 리뷰 체크리스트 (줄)
│   ├── -. # 리팩토링 패턴 (줄)
│   └── -.       # 보안 감사 체크리스트 (줄)
└── -/
    ├── -.        # 안티패턴 카탈로그 (줄)
    └── .              # 코드 메트릭 기준 (줄)
