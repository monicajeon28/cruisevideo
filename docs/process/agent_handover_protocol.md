### 인계 전 검증 (현재 )

# .  체크리스트 확인
$  - "" //.
# →    개 중 개 체크

# .   검증 실행
$  /. -- 
# →  / 

# . . 업데이트
$  /. --  --  -- 

# .   확인
$   --   -
# →      .  구현


### 인계 후 검증 (다음 )

# . . 읽기
$  - "  (.(('.'))'''')"
# → .  구현 완료...

# . 블로커 확인
$  - "  ((.(('.'))''))"
# →  (블로커 없음)

# . 이전   확인
$  - /.
# → ------      월   /.
